Table users {
  id int [primary key]
  email varchar
  password varchar
  role varchar
}

Table meta_item_categories {
  id int
  category_name varchar
}

Table meta_item_subcategories{
  id int [primary key]
  category_id int
  subcategory_name varchar
}
Ref: meta_item_subcategories.category_id > meta_item_categories.id


Table meta_item_attributes{
  id int [primary key]
  attribute_name varchar
  subcategory_id int
}
Ref: meta_item_attributes.subcategory_id > meta_item_subcategories.id


Table items{
  id int [primary key]
  name varchar
  category_id int
  subcategory_id int
}
Ref: items.category_id > meta_item_categories.id
Ref: items.subcategory_id > meta_item_subcategories.id

Table item_attributes{
  item_id int 
  attribute_id int
  attribute_value varchar
}
Ref: item_attributes.item_id > items.id
Ref: item_attributes.attribute_id > meta_item_attributes.id


Table auctions {
  id integer [primary key]
  item_id int
  seller_id int
  initial_price real
  min_increment real
  min_price real
  opening_time timestamp
  closing_time timestamp
}
Ref: auctions.item_id > items.id
Ref: auctions.seller_id > users.id



Table bidders{
  bidder_id int [primary key]
  auction_id int
  upper_limit real
  auto_bid bool
}
Ref: bidders.bidder_id > users.id
Ref: bidders.auction_id > auctions.id

Table bids{
  id int [primary key]
  auction_id int
  bidder_id int
  bid_value real
  bid_active bool
}
Ref: bids.auction_id > auctions.id
Ref: bids.bidder_id > bidders.bidder_id


Table auction_status{
  auction_id int
  status varchar // e.g "Open", "Sold", "Expired"
  winning_bid_id int
}
Ref: auction_status.auction_id > auctions.id
Ref: auction_status.winning_bid_id > bids.id


Table user_questions{
  id int [primary key]
  asker_id int
  replier_id int
  question_text varchar
  reply_text varchar
  is_open bool
}
Ref: user_questions.asker_id > users.id
Ref: user_questions.replier_id > users.id


Table alerts{
  id int [primary key]
  item_category_id int
  item_subcategory_id int
  user_id int
}
Ref: alerts.user_id > users.id
Ref: alerts.item_category_id > meta_item_categories.id
Ref: alerts.item_subcategory_id > meta_item_subcategories.id
